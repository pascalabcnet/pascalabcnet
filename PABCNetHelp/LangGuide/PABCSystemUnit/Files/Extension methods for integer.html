Методы расширения типа integer

function Between(Self: integer; a, b: integer; inclusive: boolean := true): boolean;
        Возвращает True если значение находится между двумя другими
function Clamp(Self: integer; bottom,top: integer): integer;
        Возвращает число, ограниченное диапазоном от bottom до top включительно
function ClampBottom(Self: integer; bottom: integer): integer;
        Возвращает число, ограниченное величиной bottom снизу
function ClampTop(Self: integer; top: integer): integer;
        Возвращает число, ограниченное величиной top сверху
function Divs(Self,d: integer): boolean;
        Возвращает True если целое делится на указанное значение
function DivsAll(Self: integer; params a: array of integer): boolean;
        Возвращает True если целое делится на все значения
function DivsAny(Self: integer; params a: array of integer): boolean;
        Возвращает True если целое делится на одно из значений
function Downto(Self: integer; n: integer): sequence of integer;
        Генерирует последовательность целых от текущего значения до n в убывающем порядке
function InRange(Self: integer; a,b: integer): boolean;
        Возвращает True если значение находится в диапазоне [a,b]
function IsEven(Self: integer): boolean;
        Возвращает, является ли целое четным
function IsOdd(Self: integer): boolean;
        Возвращает, является ли целое нечетным
function NotDivs(Self,d: integer): boolean; > 0;
        Возвращает True если целое не делится на указанное значение
function Range(Self: integer): sequence of integer;
        Возвращает последовательность чисел от 1 до данного
function Sqr(Self: integer): int64;
        Возвращает квадрат числа
function Sqrt(Self: integer): real;
        Возвращает квадратный корень числа
function Times(Self: integer): sequence of integer;
        Возвращает последовательность целых 0,1,...n-1
function To(Self: integer; n: integer): sequence of integer;
        Генерирует последовательность целых от текущего значения до n
